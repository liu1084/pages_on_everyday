### Install development tools on debian

```shell
cp sources.list /etc/apt/sources.list

apt-get update -y \
&& apt upgrade -y \
&& apt-get install -y \
build-essential \
libssl-dev \
openssl \
libncurses-dev \
git \
bison \
cmake \
libwrap0-dev \
zlib1g-dev \
ca-certificates \
g++ \
python-dev \
autotools-dev \
libicu-dev \
libbz2-dev \
libboost-all-dev \
pkg-config
```
